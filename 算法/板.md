```cpp
template<typename T, auto f>
struct SparseTable{
    int n;
    std::vector<std::vector<T>>dp;

    SparseTable() : n(0) {};
    SparseTable(const std::vector<T> &a){
        n = a.size();
        int siz = std::__lg(n) + 1;
        dp.assign(n, std::vector<T>(siz + 1));
        for(int i = 0; i < n; i++){
            dp[i][0] = a[i];
        }

        for(int j = 1; 1 << j <= n; j++){
            for(int i = 0; i + (1 << j) - 1 < n; i++){
                dp[i][j] = f(dp[i][j - 1], dp[i + (1 << j - 1)][j - 1]);
            }
        }
    }

    constexpr T query(int l, int r){
        int k = std::__lg(r - l + 1);
        return f(dp[l][k], dp[r - (1 << k) + 1][k]);
    }
};
```

```cpp
template<typename T>
struct Fenwick{
    std::vector<T>a;
    int n;

    Fenwick(int n_ = 0){
        n = n_ + 1;
        init();
    }

    void init(){
        a.assign(n, T{});
    }

    void add(int pos, T k){
        for(int i = pos; i < n; i += i & -i){
            a[i] += k;
        }
    }

    T sum(int pos){
        T ans{};
        for(int i = pos; i; i -= i & -i){
            ans += a[i];
        }
        return ans;
    }

    T rangesum(int l, int r){
        if(!l){
            return sum(r);
        }
        else{
            return sum(r) - sum(l - 1);
        }
    }

    int kth(T k){
        int x=0;
        for(int i = 1 << std::__lg(n); i; i >>= 1){
            if(x + i <= n && a[x + i] < k){
                k -= a[x + i];
                x += i;
            }
        }
        return x + 1;
    }
};
```

```cpp
struct DSU{
    std::vector<int>fa, rnk;
    DSU(){};
    DSU(int n){
        init(n);
    };  

    void init(int n){
        fa.resize(n + 1);
        std::iota(fa.begin(), fa.end(), 0);
        rnk.assign(n + 1, 1);
    }

    int find(int u){
        while(u != fa[u]){
            u = fa[u] = fa[fa[u]];
        }
        return u;
    }

    bool same(int u, int v){
        return find(u) == find(v);
    }

    void merge(int u, int v){
        u = find(u);
        v = find(v);
        if(rnk[u] <= rnk[v]){
            fa[u] = v;
        }else{
            fa[v] = u;
        }
        if(rnk[u] == rnk[v] && u != v){
            rnk[v]++;
        }

    }

    int rank(int u){
        return rnk[find(u)];
    }
};
```

```cpp
template<class Info>
struct SegmentTree{
    int n;
    std::vector<Info>info;

    SegmentTree():n(0) {};
    SegmentTree(int n_, Info v_ = Info()){
        init(std::vector<Info>(n_, v_));
    }

    template<class T>
    SegmentTree(std::vector<T>init_){
        init(init_);
    }

    template<class T>
    void init(std::vector<T>init_){
        n = init_.size();
        info.assign(4 << std::__lg(n), Info());
        auto build = [&](auto&self, int l, int r, int p) -> void {
            if(l == r){
                info[p] = init_[l];
                return;
            }
            int mid = l + r >> 1;
            self(self, l, mid, p << 1);
            self(self, mid + 1, r, p << 1 | 1);
            pull(p);
        };
        build(build, 0, n - 1, 1);
    }

    void pull(int p){
        info[p] = info[p << 1] + info[p << 1 | 1];
    }

    void modify(int l, int r, int p, int pos, const Info & v){
        if(l == r){
            info[p] = v;
            return;
        }
        int mid = l + r >> 1;
        if(pos <= mid){
            modify(l, mid, p << 1, pos, v);
        }else{
            modify(mid + 1, r, p << 1 | 1, pos, v);
        }
        pull(p);
    }

    void modify(int pos, const Info & v){
        modify(0, n - 1, 1, pos, v);
    }
    
    template<typename T>
    void add(int l, int r, int p, int pos, T x){
        if(l == r){
            info[p] += x;
            return;
        }
        int mid = l + r >> 1;
        if(pos<=mid){
            add(l, mid, p << 1, pos, x);
        }else{
            add(mid + 1, r, p << 1 | 1, pos, x);
        }
        pull(p);
    }

    template<typename T>
    void add(int pos, T x){
        add(0, n - 1, 1, pos, x);
    }
    
    Info query(int l, int r, int x, int y, int p){
        if(l > y || r < x){
            return Info();
        }
        if(l >= x && r <= y){
            return info[p];
        }
        int mid = l + r >> 1;
        return query(l, mid, x, y, p << 1) + query(mid + 1, r, x, y, p << 1 | 1);
    }

    Info query(int l, int r){
        return query(0, n - 1, l, r, 1);
    }
};
```

```cpp
template<class Info,class Tag>
struct LazySegmentTree{
    int n;
    std::vector<Info>info;
    std::vector<Tag>tag;

    LazySegmentTree() : n(0) {};
    LazySegmentTree(int n_, Info v = Info()){
        init(vector<Info>(n_, v));
    }

    template<class T>
    LazySegmentTree(std::vector<T>init_){
        init(init_);
    }

    template<class T>
    void init(std::vector<T>init_){
        n = init_.size();
        info.assign(4 << std::__lg(n), Info());
        tag.assign(4 << std::__lg(n), Tag());
        auto build = [&](auto&self, int l, int r, int p) -> void {
            if(l == r){
                info[p] = init_[l];
                return;
            }
            int mid = l + r >> 1;
            self(self, l, mid, p << 1);
            self(self, mid + 1, r, p <<1 | 1);
            pull(p);
        };
        build(build, 0, n-1, 1);
    }

    void pull(int p){
        info[p] = info[p << 1] + info[p << 1 | 1];
    }

    void apply(int p, const Tag &v){
        info[p].apply(v);
        tag[p].apply(v);
    }

    void push(int p){
        apply(p << 1, tag[p]);
        apply(p << 1 | 1, tag[p]);
        tag[p] = Tag();
    }

    void modify(int l, int r, int pos, int p, const Info &v){
        if(l == r){
            info[p] = v;
            return;
        }
        push(p);
        int mid = l + r >> 1;
        if(pos <= mid) {
            modify(l, mid, pos, p << 1, v);
        }
        else{
            modify(mid + 1, r, pos, p << 1 | 1, v);
        }
        pull(p);
    }

    void modify(int pos, const Info &v){
        modify(0, n - 1, pos, 1, v);
    }

    void rangeapply(int l, int r, int x, int y, int p, const Tag &v){
        if(l > y || r < x){
            return;
        }
        if(l >= x && r <= y){
            apply(p, v);
            return;
        }
        push(p);
        int mid = l + r >> 1;
        if(x <= mid){
            rangeapply(l, mid, x, y, p << 1, v);
        }
        if(y > mid){
            rangeapply(mid + 1, r, x, y, p << 1 | 1, v);
        }
        pull(p);
    }

    void rangeapply(int l, int r, const Tag &v){
        rangeapply(0, n - 1, l, r, 1, v);
    }

    Info query(int l,int r,int x,int y,int p){
        if(l > y || r < x){
            return Info();
        }
        if(l >= x && r <= y){
            return info[p];
        }
        int mid = l + r >> 1;
        push(p);
        return query(l, mid, x, y, p << 1) + query(mid + 1, r, x, y, p << 1 | 1);
    }

    Info query(int l, int r){
        return query(0, n - 1, l, r, 1);
    }

};
```

```cpp
    int n = s.size();
    std::vector<int>z(n + 1);
    z[0] = n;

    for(int i = 1, j = 1; i < n; i++){
        z[i] = std::max(0, std::min(j + z[j] - i, z[i - j]));
```

```cpp
constexpr i64 safeMod(i64 x, i64 m) {
    x %= m;
    if (x < 0) {
        x += m;
    }
    return x;

constexpr std::pair<i64, i64> invGcd(i64 a, i64 b) {
    a = safeMod(a, b);
    if (a == 0) {
        return {b, 0};
    }
    
    i64 s = b, t = a;
    i64 m0 = 0, m1 = 1;
 
    while (t) {
        i64 u = s / t;
        s -= t * u;
        m0 -= m1 * u;
        
        std::swap(s, t);
        std::swap(m0, m1);
    }
    
    if (m0 < 0) {
        m0 += b / s;
    }
    
    return {s, m0};
}

struct ModIntBase {
public:
    constexpr ModIntBase() : x(0) {}
    template<std::unsigned_integral T>

    template<std::signed_integral T>

        S v = x_ % S(mod());

            v += mod();

struct Comb {

        _invfac.resize(m + 1);
```

```cpp
            std::swap(u, v);
        }

            u = fa[u][std::__lg(dep[u] - dep[v])];

                v = fa[v][i];
```

```cpp
constexpr std::pair<i64, i64> invGcd(i64 a, i64 b) {
    a = safeMod(a, b);
    if (a == 0) {
        return {b, 0};
    }
    
    i64 s = b, t = a;
    i64 m0 = 0, m1 = 1;
 
    while (t) {
        i64 u = s / t;
        s -= t * u;
        m0 -= m1 * u;
        
        std::swap(s, t);
        std::swap(m0, m1);
    }
    
    if (m0 < 0) {
        m0 += b / s;
    }
    
    return {s, m0};
}
```
